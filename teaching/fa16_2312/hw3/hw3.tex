			\item $\frac{dx}{dt}=\frac{e^x\sin^2(t)}{x\sec{t}}$
			\item $\frac{dy}{dx}=2xy-2y+2x-2$, $y(1)=0$
			\item $x^2\frac{dy}{dx}=\sqrt{1-y^2}$
			\item $e^y\left(\frac{dy}{dx}\right)=1+e^{2y}-xe^{2y}-x$, $y(0)=1$

		\item \begin{enumerate}
			\item \label{num:3a} Write the differential equation modeling the following scenario: \textit{The rate of growth of a population $P$ over time is directly proportional to the population.}

		\item \begin{enumerate}
			\item \label{num:4a} Let $M$ be a constant and let $k$ denote a constant of proportionality. Show that the solution to the logistic differential equation
			$$\frac{dP}{dt}=kP\left(1-\frac{P}{M}\right)$$
			has the form
			$$P(t)=M\,\frac{Ce^{kt}}{1+Ce^{kt}}.$$
			\item Write the solution of the initial value problem
			$$\frac{dP}{dt}=0.08P\left(1-\frac{P}{1000}\right)\quad\quad P(0)=100.$$
			\item Show that if $P$ satisfies the logistic equation in \ref{num:4a}, then the second derivative $\frac{d^2P}{dt^2}$ satisfies the following:
			$$\frac{d^2P}{dt^2}=k^2P\left(1-\frac{P}{M}\right)\left(1-\frac{2P}{M}\right).$$

	\item Solve each of the following linear differential equations and/or linear DE IVPs.

		\item $xy'=e^x-y$, $y(1)=0$
		\item $y'=\frac{y}{x}+x$, $y(1)=1$
		\item $(1+t^2)y'+4ty=(1+t^2)^{-2}$

	\item A rocket with initial mass $M_0$ (kilograms) blasts off at time $t=0$, and after taking off, the mass of the rocket decreases with time because the fuel is being spent at a constant burning rate $r$ (kg per sec). If thrust is a constant force $F$ and velocity is $v$, Newton's second law gives
	\begin{equation}
	\label{eq:rocket}
	\frac{d}{dt}\left((M_0-rt)v\right)=F-(M_0-rt)g\end{equation}
	where $g=9.8$ is the gravitational constant. \begin{enumerate*}\item Solve the equation \eqref{eq:rocket} for $v=v(t)$, given that $v=0$ when $t=0$. \item If the mass of the rocket at burnout is $M_0-rt=M_1$, compute the velocity of the rocket at burnout. \vspace{3mm}
	\end{enumerate*}
	
	
	\textbf{Hint}: Once you have the solution in part (a), replace $M_0-rt$ by $M_1$ \textit{and} replace $rt$ by the appropriate expression in terms of $M_0$ and $M_1$ obtained by solving the equation $M_0-rt=M_1$ for $rt$.
\end{enumerate}
